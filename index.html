// Example: Hello, World!
using System;

class Program {
    static void Main() {
        Console.WriteLine("Hello, World!");
    }
}
// Exercise: Simple Addition
using System;

class Program {
    static int Add(int a, int b) {
        return a + b;
    }

    static void Main() {
        Console.WriteLine(Add(3, 5));  // Expected output: 8
    }
}
